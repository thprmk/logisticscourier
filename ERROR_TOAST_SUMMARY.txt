================================================================================
    USAGE EXAMPLES
================================================================================

EXAMPLE 1: Simple Error Handling
  import { Toasts } from '@/lib/toastHelpers';
  
  try {
    const response = await fetch('/api/shipments', { ... });
    if (!response.ok) throw new Error('Failed');
    Toasts.shipment.created();
  } catch (error) {
    Toasts.shipment.createFailed();
  }

EXAMPLE 2: Update Loading Toast
  const toastId = Toasts.shipment.creating();
  try {
    // ... operation ...
    Toasts.update(toastId, 'Shipment created!', 'success');
  } catch (error) {
    Toasts.shipment.createFailed(toastId);
  }

EXAMPLE 3: Validation Errors
  if (!email) {
    Toasts.validation('Email', 'required');
    return;
  }
  if (!isValidEmail(email)) {
    Toasts.validation('Email', 'invalid');
    return;
  }

EXAMPLE 4: HTTP Status Handling
  if (!response.ok) {
    Toasts.status(response.status); // Auto-selects message
  }
