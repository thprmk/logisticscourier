   </div>

================================================================================
                        RESPONSIVE PATTERNS USED
================================================================================

PATTERN 1: Stacked-to-Grid
──────────────────────────
Used in: Form sections, component grids

<div className="grid grid-cols-1 md:grid-cols-2 gap-4">
  {/* Mobile: 1 column, Tablet: 2 columns */}
</div>

PATTERN 2: Flex Direction Change
─────────────────────────────────
Used in: Navigation, filter bars

<div className="flex flex-col md:flex-row gap-2 md:gap-4">
  {/* Column on mobile, row on tablet+ */}
</div>

PATTERN 3: Hidden-Responsive Elements
──────────────────────────────────────
Used in: Navigation, advanced filters

<span className="hidden md:inline">Full Text</span>
<span className="md:hidden">Short</span>

PATTERN 4: Responsive Scaling
──────────────────────────────
Used in: Text, padding, margins

<p className="text-xs md:text-sm lg:text-base">
  Scales with screen size
</p>

<div className="p-4 md:p-6 lg:p-8">
  Responsive padding
</div>

PATTERN 5: Touch-Friendly Sizing
─────────────────────────────────
Used in: Buttons, inputs, interactive elements

<Button className="py-2 md:py-3 px-3 md:px-4">
  {/* 44px+ minimum tap target */}
</Button>

PATTERN 6: Viewport-Relative Sizing
────────────────────────────────────
Used in: Modals, overlays, full-width elements

<DialogContent className="w-[90vw] md:w-full max-w-md">
  {/* 90% viewport width on mobile */}
</DialogContent>

================================================================================
                        BREAKPOINT REFERENCE
================================================================================

DEFAULT (Mobile-first)
  • Screen: 0px - 640px
  • Applies to: Mobile phones
  • Default styles without prefix
  
EXAMPLE:
  text-sm p-4     /* Applied to all screens */

──────────────────────────────────────────────────────────

SM (640px+)
  • Screen: 640px - 768px
  • Applies to: Large phones, small tablets
  • Prefix: sm:
  
EXAMPLE:
  text-sm sm:text-base  /* sm: on 640px+ */
  w-full sm:w-80        /* Full width mobile, fixed on 640px+ */

──────────────────────────────────────────────────────────

MD (768px+)
  • Screen: 768px - 1024px
  • Applies to: Tablets, small laptops
  • Prefix: md:
  • Most common for showing/hiding elements
  
EXAMPLE:
  hidden md:block        /* Hidden < 768px, shown 768px+ */
  grid-cols-1 md:grid-cols-2

──────────────────────────────────────────────────────────

LG (1024px+)
  • Screen: 1024px - 1280px
  • Applies to: Laptops, desktop monitors
  • Prefix: lg:
  
EXAMPLE:
  md:w-2/3 lg:w-1/2
  grid-cols-2 md:grid-cols-3 lg:grid-cols-4

──────────────────────────────────────────────────────────

XL (1280px+)
  • Screen: 1280px - 1536px
  • Applies to: Large desktop monitors
  • Prefix: xl:
  
──────────────────────────────────────────────────────────

2XL (1536px+)
  • Screen: 1536px+
  • Applies to: Ultra-wide displays
  • Prefix: 2xl:
